从状态

1 1 1 1

转换至正确密码需要2步

从状态

1 2 3 4

转换至正确密码需要8步

求密码

注：9可以转到0

这题的答案好像就是他的手机密码（题其实被我改过了，所以别相信算出来的结果）

哎呀好难，手算算不出来一个可能解，怎么办？

额，我们不是有电脑暴力吗。

使用极为暴力的枚举算法，写出来大概这样

  1. #include <cstdio>
  2. #include <cstdlib>
  3. #define scanf scanf_s
  4. #define Max(x, y) (x > y ? x : y)
  5. #define Min(x, y) (x < y ? x : y)
  6. #define Differ(x, y) (Max(x, y) - Min(x, y))
  8. const int ttmax(10), condmax(10);
  10. int roll[condmax][ttmax], tt, cond; // 条件
  11. int con[condmax]; // 条件结果
  13. int mas[ttmax], anstot(0);
  15. int turn(int a, int up)
  16. {
  17.     a += up;
  18.     a = (a + 10) % 10;
  19.     return a;
  20. }
  22. void doit(int now, int lft) //递归枚举
  23. {
  24.     if (now == tt)
  25.     {
  26.         int ans[condmax];
  27.         for (int i(0); i < condmax; ++i) ans[i] = 0;
  28.         for (int j(0); j < cond - 1; ++j)
  29.         {
  30.             for (int i(0); i < tt; ++i)
  31.             {
  32.                 ans[j] += Min(Differ(roll[j][i], mas[i]), 10 - Differ(roll[j][i], mas[i]));
  33.             }
  34.         }
  35.         bool judge(true);
  36.         for (int i(0); i < cond - 1; ++i)
  37.         {
  38.             if (ans[i] != con[i])
  39.             {
  40.                 judge = false;
  41.             }
  42.         }
  43.         if (judge)
  44.         {
  45.             ++anstot;
  46.             for (int i(0); i < tt; ++i)
  47.             {
  48.                 printf("%d", mas[i]);
  49.             }
  50.             printf("\n");
  51.         }
  52.     }
  53.     else 
  54.     for (int i((lft - (tt - now - 1) * 5) < 0 ? 0 : (lft - (tt - now - 1) * 5)); i <= lft && i < 6; ++i)
  55.     {
  56.         mas[now] = turn(roll[cond - 1][now], i);
  57.         doit(now + 1, lft - i);
  58.         if (i != 0 && i != 5)
  59.         {
  60.             mas[now] = turn(roll[cond - 1][now], -i);
  61.             doit(now + 1, lft - i);
  62.         }
  63.     }
  64. }
  66. int main()
  67. {
  68.     scanf("%d%d", &tt, &cond); // 输入位数，条件个数
  69.     for (int i(0); i < cond; ++i) // 输入条件
  70.     {
  71.         for (int j(0); j < tt; ++j)
  72.         {
  73.             scanf("%d", &roll[i][j]);
  74.         }
  75.         scanf("%d", &con[i]);
  76.     }
  77.     doit(0, con[cond - 1]);
  78.     printf("%d\n", anstot); // 输出解总个数
  79.     system("pause");
  80.     return 0;
  81. }
