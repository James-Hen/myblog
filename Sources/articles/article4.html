用c++实现高斯消元，步骤差不多

我用了一个类来封装矩阵，高斯消元函数为Gauss();

这个模板类的模板代表矩阵元素的类型

  1. #pragma once
  3. #include <iostream>
  5. using namespace std;
  7. namespace James
  8. {
  10.     template<typename T>
  11.     class matrix
  12.     {
  13.     private:
  14.         T * * root;
  15.         int lnes, cols;
  16.     public:
  17.         matrix(int m, int n); //构造
  18.         T * operator [] (int ind); //定义方括号方便取元素
  19.         void print(); //输出
  20.         void Gauss(); //消元
  21.     };
  23.     template<typename T>
  24.     inline matrix<T>::matrix(int m, int n) : lnes(m), cols(n)
  25.     {
  26.         root = new T * [m];
  27.         for (int i(0); i < m; ++i)
  28.         {
  29.             root[i] = new T [n];
  30.         }
  31.     }
  33.     template<typename T>
  34.     inline T * matrix<T>::operator[](int ind)
  35.     {
  36.         return root[ind];
  37.     }
  39.     template<typename T>
  40.     void James::matrix<T>::print()
  41.     {
  42.         for (int i(0); i < 80; ++i) cout << '-';
  43.         cout << endl;
  44.         for (int i(0); i < lnes; ++i)
  45.         {
  46.             cout << "|\t";
  47.             for (int j(0); j < cols; ++j)
  48.             {
  49.                 cout << root[i][j] << "\t";
  50.             }
  51.             cout << "\t|\n";
  52.         }
  53.         for (int i(0); i < 80; ++i) cout << '-';
  54.         cout << endl;
  55.     }
  57.     template<typename T>
  58.     void James::matrix<T>::Gauss()
  59.     {
  60.         for (int i(0); i < lnes; ++i)
  61.         {
  62.             for (int j(i + 1); j < cols; ++j)
  63.             {
  64.                 root[i][j] /= root[i][i];
  65.             }
  66.             root[i][i] = 1;
  67.             for (int x(i + 1); x < lnes; ++x)
  68.             {
  69.                 for (int j(i + 1); j < cols; ++j)
  70.                 {
  71.                     root[x][j] -= root[x][i] / root[i][i] * root[i][j];
  72.                 }
  73.                 root[x][i] = 0;
  74.             }
  75.         } // 往下推的过程
  76.         for (int i(lnes - 1); i >= 0; --i)
  77.         {
  78.             for (int j(0); j < i; ++j)
  79.             {
  80.                 root[j][cols - 1] -= root[j][i] / root[i][i] * root[i][cols - 1];
  81.                 root[j][i] = 0;
  82.             }
  83.         } //回代的过程
  84.     }
  86. }
